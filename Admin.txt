import java.util.*;

public class Admin {
    private String id;
    private String password;

    public Admin(String id, String password) {
        this.id = id;
        this.password = password;
    }

    public boolean login(String id, String password) {
        return this.id.equals(id) && this.password.equals(password);
    }

    public void addStudent(ArrayList<Student> students, Scanner sc) {
        System.out.print("Enter student name: ");
        String name = sc.nextLine();
        System.out.print("Enter age: ");
        int age = sc.nextInt(); sc.nextLine();
        System.out.print("Enter email: ");
        String email = sc.nextLine();
        System.out.print("Enter gender: ");
        String gender = sc.nextLine();
        System.out.print("Set student ID: ");
        String stid = sc.nextLine();
        System.out.print("Set student password: ");
        String pw = sc.nextLine();
        students.add(new Student(name, age, email, gender, stid, pw));
        System.out.println("Student added and login created!");
    }

    public void registerStudentToCourse(ArrayList<Student> students, ArrayList<Course> courses, Scanner sc) {
        System.out.print("Enter student ID: ");
        String id = sc.nextLine();
        Student s = null;
        for (Student stud : students) {
            if (stud.id.equals(id)) s = stud;
        }
        if (s == null) { System.out.println("Student not found!"); return; }
        System.out.print("Enter course code: ");
        String code = sc.nextLine();
        Course c = null;
        for (Course course : courses) {
            if (course.code.equals(code)) c = course;
        }
        if (c == null) { System.out.println("Course not found!"); return; }
        s.courses.add(c);
        System.out.println("Student registered to course.");
    }

    public void viewAllStudents(ArrayList<Student> students) {
        for (Student s : students) s.display();
    }

    public void addCourse(ArrayList<Course> courses, Scanner sc) {
        System.out.print("Enter new course code: ");
        String code = sc.nextLine();
        System.out.print("Enter course name: ");
        String name = sc.nextLine();
        courses.add(new Course(code, name));
        System.out.println("Course added.");
    }
}
